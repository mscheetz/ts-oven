import { Pool } from 'pg';

class PGBaseRepo {
    private pool: Pool;

    constructor() {
        this.pool = new Pool({
            user: process.env.PGUSER,
            host: process.env.PGHOST,
            database: process.env.PGDATABASE,
            password: process.env.PGPASSWORD,
            port: process.env.PGPORT,
            ssl: { rejectUnauthorized: false }
        });
    }
    
    public get = async(id: string) => {
        const sql = `select id, name
        from public."base"
        where id = $1`;

        try {
            const res = await this.pool.query(sql, [ id ]);

            return res.rows;
        } catch(err) {
            console.log(err);

            return null;
        }
    }
    
    public getAll = async() => {
        const sql = `select id, name
        from public."base"`;

        try {
            const res = await this.pool.query(sql);

            return res.rows;
        } catch(err) {
            console.log(err);
            
            return null;
        }
    }
    
    public add = async(base: Base) => {
        const sql = `insert into public."base" ( id, name )
        values ( $1, $2 )`;

        const data = [
            base.id,
            base.name
        ];

        try {
            const res = await this.pool.query(sql, data);

            return res.rowCount;
        } catch(err) {
            console.log(err);
            
            return null;
        }
    }
    
    public update = async(base: Base) => {
        const sql = `update public."base" set name = $2
        where id = $1`;

        const data = [
            base.id,
            base.name
        ];

        try {
            const res = await this.pool.query(sql, data);

            return res.rowCount;
        } catch(err) {
            console.log(err);
            
            return null;
        }
    }
    
    public delete = async(id: string) => {
        const sql = `delete from  public."base"
        where id = $1`;

        try {
            const res = await this.pool.query(sql, [ id ]);

            return res.rowCount;
        } catch(err) {
            console.log(err);
            
            return null;
        }
    }

}

export default PGBaseRepo;
